def extra_attribute(name, options = {}, &blk)
  allow_field = proc {
    if options[:if]
      next false unless instance_eval(&options[:if])
    end

    @extra_fields[@_type] && @extra_fields[@_type].include?(name)
  }

  attribute name, if: allow_field, &blk
end
